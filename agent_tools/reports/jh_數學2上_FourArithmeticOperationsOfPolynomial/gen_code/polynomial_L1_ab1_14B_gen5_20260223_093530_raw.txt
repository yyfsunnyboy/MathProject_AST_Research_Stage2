import random

def generate(level=1, **kwargs):
    ops = ['+', '-', '*', '/']
    op = random.choice(ops)
    if op in ['+', '-']:
        poly1 = generate_poly(level)
        poly2 = generate_poly(level)
        if op == '+':
            correct = add_polynomials(poly1, poly2)
        else:
            correct = subtract_polynomials(poly1, poly2)
        question = f"計算 ${poly1}$ {op} ${poly2}$"
    elif op == '*':
        poly1 = generate_poly(level)
        poly2 = generate_poly(level)
        correct = multiply_polynomials(poly1, poly2)
        question = f"展開並化簡 ${poly1}$ ${poly2}$"
    else:
        poly1 = generate_poly(level)
        poly2 = generate_poly(level)
        while not divisible_polynomials(poly1, poly2):
            poly2 = generate_poly(level)
        correct = divide_polynomials(poly1, poly2)
        question = f"計算 ${poly1}$ ÷ ${poly2}$"
    return {
        'question_text': question,
        'answer': '',
        'correct_answer': correct,
        'mode': 1
    }

def check(user_answer, correct_answer):
    correct = str(user_answer).strip() == str(correct_answer).strip()
    return {'correct': correct, 'result': '正確' if correct else '錯誤'}

def generate_poly(level):
    terms = []
    max_degree = 2 + level
    max_coeff = 5 + level*2
    num_terms = random.randint(1, 3)
    for _ in range(num_terms):
        coeff = random.randint(-max_coeff, max_coeff)
        if coeff == 0:
            continue
        degree = random.randint(0, max_degree)
        if degree == 0:
            term = f"{coeff}"
        else:
            term = f"{coeff}x"
            if degree > 1:
                term += f"^{degree}"
        terms.append(term)
    return ' + '.join(terms).replace('+ -', '-')

def add_polynomials(p1, p2):
    return combine_terms(p1 + ' + ' + p2)

def subtract_polynomials(p1, p2):
    return combine_terms(p1 + ' - ' + p2)

def multiply_polynomials(p1, p2):
    terms = []
    p1_terms = p1.split(' + ')
    p2_terms = p2.split(' + ')
    for t1 in p1_terms:
        for t2 in p2_terms:
            terms.append(multiply_terms(t1, t2))
    return combine_terms(' + '.join(terms))

def divide_polynomials(p1, p2):
    return combine_terms(p1 + ' ÷ ' + p2)
