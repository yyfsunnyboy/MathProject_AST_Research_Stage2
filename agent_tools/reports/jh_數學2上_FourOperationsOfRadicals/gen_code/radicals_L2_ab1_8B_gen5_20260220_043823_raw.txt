import random
import math

def generate(level=1, **kwargs):
    square_factors = [4, 9, 25, 49]
    coefficients = [2, 3, 5, 7]
    terms_part1 = []
    for _ in range(random.randint(2, 4)):
        sign = random.choice([1, -1])
        square_factor = random.choice(square_factors)
        coefficient_under_sqrt = random.choice(coefficients)
        number = square_factor * coefficient_under_sqrt
        term = f"{sign * random.randint(1, 3)} \\sqrt{{{number}}}"
        terms_part1.append(term)
    part1 = " + ".join(terms_part1)
    
    part2_terms = []
    for _ in range(2):
        sign = random.choice([1, -1])
        square_factor = random.choice(square_factors)
        coefficient_under_sqrt = random.choice(coefficients)
        number = square_factor * coefficient_under_sqrt
        term = f"{sign * random.randint(1, 3)} \\sqrt{{{number}}}"
        part2_terms.append(term)
    part2 = " + ".join(part2_terms)
    part2 = f"({part2})"
    
    question_text = f"({part1}) + ({part2})"
    
    def simplify_term(a, b):
        s = 1
        for i in range(2, int(math.isqrt(b)) + 1):
            if b % (i*i) == 0:
                s = i*i
        k = b // s
        simplified_coeff = a * math.isqrt(s)
        return f"{simplified_coeff}\\sqrt{{{k}}}"
    
    def combine_terms(terms):
        from collections import defaultdict
        grouped = defaultdict(int)
        for term in terms:
            parts = term.split()
            coeff = int(parts[0])
            radical = parts[2]
            key = radical
            grouped[key] += coeff
        result = []
        for key in sorted(grouped):
            result.append(f"{grouped[key]}\\sqrt{{{key[2:]}}}")
        return " + ".join(result)
    
    terms = []
    for term in terms_part1:
        parts = term.split()
        coeff = int(parts[0])
        number = int(parts[2][2:])
        simplified = simplify_term(coeff, number)
        terms.append(simplified)
    part2_terms = []
    for term in part2_terms:
        parts = term.split()
        coeff = int(parts[0])
        number = int(parts[2][2:])
        simplified = simplify_term(coeff, number)
        part2_terms.append(simplified)
    part2 = " * ".join(part2_terms)
    part2 = f"({part2})"
    terms.append(f"({part2})")
    correct_answer = combine_terms(terms)
    
    return {
        'question_text': f'${question_text}$',
        'answer': '',
        'correct_answer': correct_answer,
        'mode': 1
    }

def check(user_answer, correct_answer):
    correct = str(user_answer).strip() == str(correct_answer).strip()
    return {'correct': correct, 'result': '正確' if correct else '錯誤'}
