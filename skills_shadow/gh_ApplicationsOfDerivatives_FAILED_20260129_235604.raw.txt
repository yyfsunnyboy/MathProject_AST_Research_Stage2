def generate(level=1, **kwargs):
    while True:
        # 1. 生成原始多項式的次數 (degree)
        degree = random.choice([3, 4, 5])

        # 2. 生成多項式係數 (polynomial_coeffs)
        # coeffs[i] 代表 x^i 的係數
        polynomial_coeffs = [0] * (degree + 1)
        
        # 確保最高次項係數非零
        polynomial_coeffs[degree] = random.choice([i for i in range(-10, 11) if i != 0])
        
        # 生成其他係數，允許為零
        for i in range(degree):
            polynomial_coeffs[i] = random.randint(-10, 10)
        
        # 檢查原始多項式是否至少有 3 個非零係數
        non_zero_count = sum(1 for c in polynomial_coeffs if c != 0)
        if non_zero_count < 3:
            continue # 不符合條件，重新生成

        # 3. 生成兩個不同的導數次數 (derivative_order_1, derivative_order_2)
        derivative_order_1 = random.randint(1, degree)
        
        derivative_order_2 = random.randint(1, degree)
        while derivative_order_2 == derivative_order_1:
            derivative_order_2 = random.randint(1, degree)
        
        # 4. 迭代計算各階導數
        current_coeffs = list(polynomial_coeffs) # 複製一份原始係數
        f_prime1_coeffs = []
        f_prime2_coeffs = []

        max_order = max(derivative_order_1, derivative_order_2)

        for i in range(1, max_order + 1):
            next_coeffs = []
            # 多項式微分規則: (c_k * x^k)' = k * c_k * x^(k-1)
            # current_coeffs[j] 是 x^j 的係數
            # next_coeffs[j-1] 將是 current_coeffs[j] * j
            for j in range(1, len(current_coeffs)): # 從 x^1 項開始 (係數 current_coeffs[1])
                next_coeffs.append(current_coeffs[j] * j)
            current_coeffs = next_coeffs
            
            if i == derivative_order_1:
                f_prime1_coeffs = list(current_coeffs)
            if i == derivative_order_2:
                f_prime2_coeffs = list(current_coeffs)

        # 5. 驗證導數結果的係數絕對值不超過 1000
        coeff_abs_check_passed = True
        for c in f_prime1_coeffs:
            if abs(c) > 1000:
                coeff_abs_check_passed = False
                break
        if not coeff_abs_check_passed:
            continue # 不符合條件，重新生成

        for c in f_prime2_coeffs:
            if abs(c) > 1000:
                coeff_abs_check_passed = False
                break
        if not coeff_abs_check_passed:
            continue # 不符合條件，重新生成
        
        # 如果所有約束都通過，則跳出循環
        break

    # 6. 格式化題幹 (LaTeX 格式)
    f_x_latex = fmt_polynomial(polynomial_coeffs, is_latex=True)

    # 根據導數次數生成 LaTeX 符號
    def get_latex_derivative_symbol(order):
        if order == 1:
            return "f'(x)"
        elif order == 2:
            return "f''(x)"
        else:
            # 使用三層大括號以正確在 f-string 中表達 LaTeX 上標
            return f"f^{{({order})}}(x)" 
    
    prime1_symbol_latex = get_latex_derivative_symbol(derivative_order_1)
    prime2_symbol_latex = get_latex_derivative_symbol(derivative_order_2)

    q = f"已知 $f(x) = {f_x_latex}$，求 ${prime1_symbol_latex}$ 與 ${prime2_symbol_latex}$。"
    q = clean_latex_output(q) # 自動為數學部分添加 $ $

    # 7. 格式化答案 (純文本格式)
    f_prime1_plain = fmt_polynomial(f_prime1_coeffs, is_latex=False)
    f_prime2_plain = fmt_polynomial(f_prime2_coeffs, is_latex=False)

    # 根據導數次數生成純文本符號
    def get_plain_derivative_symbol(order):
        if order == 1:
            return "f'(x)"
        elif order == 2:
            return "f''(x)"
        else:
            return f"f^({order})(x)" # 純文本不需要 LaTeX 語法
    
    prime1_symbol_plain = get_plain_derivative_symbol(derivative_order_1)
    prime2_symbol_plain = get_plain_derivative_symbol(derivative_order_2)

    a = f"{prime1_symbol_plain} = {f_prime1_plain}\n{prime2_symbol_plain} = {f_prime2_plain}"

    # 8. 返回結果字典
    return {
        'question_text': q,
        'correct_answer': a,
        'answer': a,
        'mode': 1
    }
